201 inline GLenum
202 toCanonicalType(GLenum type)
203 {
204  switch (type)
205  {
206  case GL_BOOL : return GL_INT;
207  case GL_BOOL_VEC2_ARB : return GL_INT_VEC2_ARB;
208  case GL_BOOL_VEC3_ARB : return GL_INT_VEC3_ARB;
209  case GL_BOOL_VEC4_ARB : return GL_INT_VEC4_ARB;
210  case GL_SAMPLER_1D_ARB: return GL_TEXTURE_1D;
211  case GL_SAMPLER_2D_ARB: return GL_TEXTURE_2D;
212  case GL_SAMPLER_3D_ARB: return GL_TEXTURE_3D;
213  case GL_SAMPLER_CUBE_ARB: return GL_TEXTURE_CUBE_MAP_ARB;
214  case GL_SAMPLER_2D_RECT_ARB: return GL_TEXTURE_RECTANGLE_ARB;
215  case GL_SAMPLER_2D_SHADOW_ARB: return GL_TEXTURE_2D;
216  case GL_SAMPLER_2D_RECT_SHADOW_ARB: return GL_TEXTURE_RECTANGLE_ARB;
217  default: return type;
218  }

248 template <typename T>
249 GLenum
250 uniformTypeFromMatrix(IDenseMatrix<T> const & m)
251 {
252  switch (m.rows())
253  {
254  case 1:
255  switch (m.cols())
256  {
257  case 2: return std::is_integral<T>::value ? GL_INT_VEC2_ARB : GL_FLOAT_VEC2_ARB;
258  case 3: return std::is_integral<T>::value ? GL_INT_VEC3_ARB : GL_FLOAT_VEC3_ARB;
259  case 4: return std::is_integral<T>::value ? GL_INT_VEC4_ARB : GL_FLOAT_VEC4_ARB;
260  }
261  break;
262 
263  case 2:
264  switch (m.cols())
265  {
266  case 1: return std::is_integral<T>::value ? GL_INT_VEC2_ARB : GL_FLOAT_VEC2_ARB;
267  case 2: return std::is_integral<T>::value ? GL_INVALID_ENUM : GL_FLOAT_MAT2_ARB;
268  }
269  break;
270 
271  case 3:
272  switch (m.cols())
273  {
274  case 1: return std::is_integral<T>::value ? GL_INT_VEC3_ARB : GL_FLOAT_VEC3_ARB;
275  case 3: return std::is_integral<T>::value ? GL_INVALID_ENUM : GL_FLOAT_MAT3_ARB;
276  }
277  break;
278 
279  case 4:
280  switch (m.cols())
281  {
282  case 1: return std::is_integral<T>::value ? GL_INT_VEC4_ARB : GL_FLOAT_VEC4_ARB;
283  case 4: return std::is_integral<T>::value ? GL_INVALID_ENUM : GL_FLOAT_MAT4_ARB;
284  }
285  break;
286  }
287 
288  return GL_INVALID_ENUM;
289 }

306 template <typename T>
307 int8
308 GlShader::setUniformHelper(char const * uniform_name, IDenseMatrix<T> const * value)
309 {
310  typedef typename GlShaderInternal::GlType<T>::type GT;
311 
312  if (!value)
313  { THEA_ERROR << getName() << ": Uniform '" << uniform_name << "' can't be assigned a null value"; return GlCaps::setError(); }
314 
315  GLenum gl_type = GlShaderInternal::uniformTypeFromMatrix(*value);
316  if (gl_type == GL_INVALID_ENUM)
317  {
318  THEA_ERROR << getName() << ": Matrix provided for uniform '" << uniform_name << "' doesn't correspond to a valid GLSL type";
319  return GlCaps::setError();
320  }
321 
322  Uniforms::iterator entry = uniforms.find(uniform_name);
323  THEA_GLSHADER_SET_UNIFORM_CHECKS(gl_type, 1)
324 
325  size_t num_elems = (size_t)(value->rows() * value->cols());
326  entry->second.value.getArray<GT>().resize(num_elems);
327  GlInternal::toArray(*value, entry->second.value.getArray<GT>().data());
328 
329  entry->second.valueChanged();
330  return true;
331 }

335 GlShader::setUniformHelper(char const * uniform_name, int64 num_values, T const * values)
336 {

338 
339  if (!values)

351 template <typename T>
352 int8
353 GlShader::setUniformHelper(char const * uniform_name, int64 num_values, IDenseMatrix<T> const * const * values)
354 {
355  typedef typename GlShaderInternal::GlType<T>::type GT;
356 
357  if (!values)
358  { THEA_ERROR << getName() << ": Uniform '" << uniform_name << "' can't be assigned a null array"; return GlCaps::setError(); }
359  if (num_values < 1)
360  { THEA_ERROR << getName() << ": Uniform '" << uniform_name << "' assigned empty array"; return GlCaps::setError(); }
361 
362  for (intx i = 0; i < num_values; ++i)
363  if (!values[0])
364  { THEA_ERROR << getName() << ": Uniform '" << uniform_name << "' assigned null array entry"; return GlCaps::setError(); }
365 
366  GLenum gl_type = GlShaderInternal::uniformTypeFromMatrix(*values[0]);
367  if (gl_type == GL_INVALID_ENUM)
368  {
369  THEA_ERROR << getName() << ": Matrix provided for uniform '" << uniform_name << "' doesn't correspond to a valid GLSL type";
370  return GlCaps::setError();
371  }
372 
373  for (intx i = 1; i < num_values; ++i)
374  if (GlShaderInternal::uniformTypeFromMatrix(*values[1]) != gl_type)
375  {
376  THEA_ERROR << getName() << ": Matrices provided for uniform '" << uniform_name << "' must have identical dimensions";
377  return GlCaps::setError();
378  }
379 
380  Uniforms::iterator entry = uniforms.find(uniform_name);
381  THEA_GLSHADER_SET_UNIFORM_CHECKS(gl_type, num_values)
382 
383  // Pack the unrolled matrices densely in the cached array
384  intx num_elems = values[0]->rows() * values[0]->cols();
385  entry->second.value.getArray<GT>().resize((size_t)(num_values * num_elems));
386  for (intx i = 0; i < num_values; ++i)
387  GlInternal::toArray(*values[i], entry->second.value.getArray<GT>().data() + i * num_elems);
388 
389  entry->second.valueChanged();
390  return true;
391 }
