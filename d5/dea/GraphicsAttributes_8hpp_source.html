298 // Dirty SFINAE template trick to check if the type T has get...() and set...() functions
299 #define THEA_HAS_GRAPHICS_ATTRIB(name, get_func_name, set_func_name) \
300 template <typename T> \
301 class name \
302 { \
303  public: \
304  typedef char Yes; \
305  typedef struct { char a[2]; } No; \
306 \
307  struct GetBaseMixin { void get_func_name() {} }; \
308  struct SetBaseMixin { void set_func_name() {} }; \
309 \
310  struct GetBase : public T, public GetBaseMixin {}; \
311  struct SetBase : public T, public SetBaseMixin {}; \
312  template <typename S, S> class Helper {}; \
313 \
314  template <typename U> static No testGet(U *, Helper<void (GetBaseMixin::*)(), &U::get_func_name> * = 0); \
315  static Yes testGet(...); \
316 \
317  template <typename U> static No testSet(U *, Helper<void (SetBaseMixin::*)(), &U::set_func_name> * = 0); \
318  static Yes testSet(...); \
319 \
320  public: \
321  static bool const has_get = (sizeof(testGet((GetBase *)(0))) == sizeof(Yes)); \
322  static bool const has_set = (sizeof(testSet((SetBase *)(0))) == sizeof(Yes)); \
323  static bool const value = has_get && has_set; \
324 };
