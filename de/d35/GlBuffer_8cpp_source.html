101 int8
102 GlBuffer::update(int64 start_value, int64 num_values_to_update, int ncomp, int32 type, GLenum gl_target_, void const * src)
103 {
104  if (num_values_to_update <= 0) return true;
105  if (!isValid()) { THEA_ERROR << "GlBuffer: Can't update invalid buffer"; return GlCaps::setError(); }
106  if (ncomp < 1)
107  { THEA_ERROR << "GlBuffer: Can't update buffer with values of dimensionality <= 0"; return GlCaps::setError(); }
108 
109  GLenum gl_type_ = GlInternal::getGlType(NumericType(type));
110  if (gl_type_ == GL_INVALID_ENUM) { THEA_ERROR << "GlBuffer: Invalid data type"; return GlCaps::setError(); }
111 
112  if (num_values > 0)
113  {
114  if (gl_type != gl_type_ || gl_target != gl_target_)
115  {
116  THEA_ERROR << "GlBuffer: Can't update non-empty buffer with values of a different type";
117  return GlCaps::setError();
118  }
119  }
120  else
121  {
122  num_components = ncomp;
123  component_type = type;
124  gl_type = gl_type_;
125  value_size = ncomp * NumericType(type).numBits() / 8;
126  gl_target = gl_target_;
127  }
128 
129  int64 offset_bytes = start_value * value_size;
130  int64 num_bytes = num_values_to_update * value_size;
131  int64 total_size = offset_bytes + num_bytes;
132  if (total_size > capacity)
133  {
134  THEA_ERROR << "GlBuffer: Can't update beyond end of buffer";
135  return GlCaps::setError();
136  }
137 
138  if (start_value + num_values_to_update > num_values)
139  num_values = start_value + num_values_to_update;
140 
141  return uploadToGraphicsSystem(offset_bytes, num_bytes, src);
142 }
