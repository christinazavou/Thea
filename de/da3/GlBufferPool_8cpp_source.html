25 {
26  if (gpu_memory && !THEA_GL_SUPPORTS(ARB_vertex_buffer_object))
27  throw Error(std::string(getName()) + ": OpenGL vertex/index buffers in GPU memory are not supported");
28 
29  assert(capacity > 0);

31  if (gpu_memory)
32  {
33  { GlClientScope scope(GL_CLIENT_VERTEX_ARRAY_BIT);
34 
35  glGenBuffersARB(1, &gl_buffer);
36  glBindBufferARB(GL_ARRAY_BUFFER_ARB, gl_buffer);
37  THEA_CHECK_GL_OK
38 
39  GLenum gl_usage;
40  switch (usage)
41  {
42  case Usage::WRITE_EVERY_FRAME: gl_usage = GL_STREAM_DRAW_ARB; break;
43  case Usage::WRITE_OCCASIONALLY: gl_usage = GL_DYNAMIC_DRAW_ARB; break;
44  case Usage::WRITE_ONCE: gl_usage = GL_STATIC_DRAW_ARB; break;
45  default: gl_usage = GL_STREAM_DRAW_ARB;
46  }
47 
48  // Load some (undefined) data to initialize the buffer
49  glBufferDataARB(GL_ARRAY_BUFFER_ARB, (GLsizei)capacity, nullptr, gl_usage);
50  THEA_CHECK_GL_OK
51 
52  base_pointer = nullptr;
53 
54  glBindBufferARB(GL_ARRAY_BUFFER_ARB, 0);
55  THEA_CHECK_GL_OK
56  }
57  }
