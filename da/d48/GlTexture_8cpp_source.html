159  depth = images[0]->getDepth();

168  {

177 
178  glGenTextures(1, &gl_id);
179  THEA_CHECK_GL_OK
180 
181  { GlScope scope(GL_TEXTURE_BIT | GL_ENABLE_BIT); // Can we do without ENABLE_BIT? The doc is unclear.
182  GlClientScope client_scope(GL_CLIENT_PIXEL_STORE_BIT);
183 
184  glEnable(gl_target);
185  glBindTexture(gl_target, gl_id);
186  THEA_CHECK_GL_OK
187

215 int8
216 GlTexture::glTexImage(void const * bytes, Format const * bytes_format, int32 face)
217 {
218  switch (gl_target)
219  {
220  case GL_TEXTURE_1D:
221  glTexImage1D(gl_target, 0, format->openGlFormat(), (GLsizei)width, 0, bytes_format->openGlBaseFormat(),
222  bytes_format->openGlDataFormat(), bytes);
223  break;
224 
225  case GL_TEXTURE_2D:
226  case GL_TEXTURE_RECTANGLE_ARB:
227  glTexImage2D(gl_target, 0, format->openGlFormat(), (GLsizei)width, (GLsizei)height, 0, bytes_format->openGlBaseFormat(),
228  bytes_format->openGlDataFormat(), bytes);
229  break;
230 
231  case GL_TEXTURE_3D:
232 #ifdef GL_VERSION_1_2 // Apple omits the EXT extension after moving 3D textures to core
233  glTexImage3D
234 #else
235  glTexImage3DEXT
236 #endif
237  (gl_target, 0, format->openGlFormat(), (GLsizei)width, (GLsizei)height, (GLsizei)depth, 0,
238  bytes_format->openGlBaseFormat(), bytes_format->openGlDataFormat(), bytes);
239  break;
240 
241  default: // GL_TEXTURE_CUBE_MAP_ARB
242  glTexImage2D(toGlCubeMapFace(face), 0, format->openGlFormat(), (GLsizei)width, (GLsizei)height, 0,
243  bytes_format->openGlBaseFormat(), bytes_format->openGlDataFormat(), bytes);
244  }
245  THEA_GL_OK_OR_RETURN(0)
246 
247  return true;

278 int8
279 GlTexture::doSanityChecks()
280 {
281  if (dimension < 0 || dimension >= Dimension::NUM)
282  { THEA_ERROR << getName() << ": Invalid dimensionality"; return GlCaps::setError(); }
283 
284  if (dimension == Dimension::DIM_CUBE_MAP && !THEA_GL_SUPPORTS(ARB_texture_cube_map))
285  { THEA_ERROR << getName() << ": Cube map textures are not supported"; return GlCaps::setError(); }
286 
287  if (dimension == Dimension::DIM_RECTANGLE && !THEA_GL_SUPPORTS(ARB_texture_rectangle))
288  { THEA_ERROR << getName() << ": Rectangular textures are not supported"; return GlCaps::setError(); }
289 
290  if (width < 1 || height < 1 || depth < 1)
291  { THEA_ERROR << getName() << ": Texture must be at least one pixel wide in each dimension"; return GlCaps::setError(); }
292 
293  if (depth > 1 && dimension != Dimension::DIM_3D)
294  { THEA_ERROR << getName() << ": Only a 3D texture can have depth greater than one pixel"; return GlCaps::setError(); }
295 
296  if (dimension == Dimension::DIM_1D && (height > 1 || depth > 1))
297  { THEA_ERROR << getName() << ": A 1D texture cannot have height or depth greater than one pixel"; return GlCaps::setError(); }
298 
299  if (!(Math::isPowerOf2(width) && Math::isPowerOf2(height) && Math::isPowerOf2(depth)) && dimension != Dimension::DIM_RECTANGLE
300  && !THEA_GL_SUPPORTS(ARB_texture_non_power_of_two)) // rectangular textures can be npot by the spec
301  { THEA_ERROR << getName() << ": Non-power-of-two textures are not supported"; return GlCaps::setError(); }
302 
303  return true;
304 }

306 int8
307 GlTexture::setOptions(Options const * options)
308 {
309  if (!options) options = TextureOptions::defaults();
310 
311  if (options->wrapMode() < 0 || options->wrapMode() >= Options::WrapMode::NUM)
312  { THEA_ERROR << getName() << ": Invalid wrap mode"; return GlCaps::setError(); }
313 
314  if (options->interpolateMode() < 0 || options->interpolateMode() >= Options::InterpolateMode::NUM)
315  { THEA_ERROR << getName() << ": Invalid interpolation mode"; return GlCaps::setError(); }
316 
317  if (options->depthReadMode() < 0 || options->depthReadMode() >= Options::DepthReadMode::NUM)
318  { THEA_ERROR << getName() << ": Invalid depth read mode"; return GlCaps::setError(); }
319 
320  if (dimension == Dimension::DIM_RECTANGLE && options->wrapMode() == Options::WrapMode::TILE) // see GL_ARB_texture_rectangle
321  { THEA_ERROR << getName() << ": Tiling is not supported for rectangular textures"; return GlCaps::setError(); }
322 
323  GLenum wrap = GL_REPEAT;
324  switch (options->wrapMode())
325  {
326  case Options::WrapMode::CLAMP: wrap = THEA_GL_SUPPORTS(EXT_texture_edge_clamp) ? GL_CLAMP_TO_EDGE : GL_CLAMP; break;
327  case Options::WrapMode::TILE: wrap = GL_REPEAT; break;
328  case Options::WrapMode::ZERO:
329  {
330  wrap = THEA_GL_SUPPORTS(ARB_texture_border_clamp) ? GL_CLAMP_TO_BORDER_ARB : GL_CLAMP;
331  GLfloat border_color[4] = {0, 0, 0, 1};
332  glTexParameterfv(gl_target, GL_TEXTURE_BORDER_COLOR, border_color);
333  break;
334  }
335  default: THEA_ERROR << getName() << ": Unsupported wrap mode"; return GlCaps::setError();
336  }
337 
338  glTexParameteri(gl_target, GL_TEXTURE_WRAP_S, wrap);
339  glTexParameteri(gl_target, GL_TEXTURE_WRAP_T, wrap);
340  glTexParameteri(gl_target, GL_TEXTURE_WRAP_R, wrap);
341 
342  THEA_GL_OK_OR_RETURN(0)
343 
344  int8 has_mipmaps = (options->interpolateMode() == Options::InterpolateMode::NEAREST_MIPMAP
345  || options->interpolateMode() == Options::InterpolateMode::BILINEAR_MIPMAP
346  || options->interpolateMode() == Options::InterpolateMode::TRILINEAR);
347 
348  if (dimension == Dimension::DIM_RECTANGLE && has_mipmaps) // see GL_ARB_texture_rectangle
349  { THEA_ERROR << getName() << ": Mipmapping is not supported for rectangular textures"; return GlCaps::setError(); }
350 
351  switch (options->interpolateMode())
352  {
353  case Options::InterpolateMode::NEAREST_NO_MIPMAP:
354  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
355  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_NEAREST);
356  break;
357 
358  case Options::InterpolateMode::NEAREST_MIPMAP:
359  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_NEAREST);
360  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_NEAREST_MIPMAP_NEAREST);
361  break;
362 
363  case Options::InterpolateMode::BILINEAR_NO_MIPMAP:
364  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
365  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR);
366  break;
367 
368  case Options::InterpolateMode::BILINEAR_MIPMAP:
369  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
370  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_NEAREST);
371  break;
372 
373  case Options::InterpolateMode::TRILINEAR:
374  glTexParameteri(gl_target, GL_TEXTURE_MAG_FILTER, GL_LINEAR);
375  glTexParameteri(gl_target, GL_TEXTURE_MIN_FILTER, GL_LINEAR_MIPMAP_LINEAR);
376  break;
377 
378  default: THEA_ERROR << getName() << ": Unsupported texture interpolation mode"; return GlCaps::setError();
379  }
380 
381  THEA_GL_OK_OR_RETURN(0)
382 
383  if (has_mipmaps)
384  {
385  if (THEA_GL_SUPPORTS(SGIS_generate_mipmap))
386  glTexParameteri(gl_target, GL_GENERATE_MIPMAP_SGIS, GL_TRUE);
387  else if (GL_VERSION_1_4) // moved to core in 1.4
388  glTexParameteri(gl_target, GL_GENERATE_MIPMAP, GL_TRUE);
389  else
390  {
391  THEA_ERROR << getName() << ": Automatic mipmap generation not supported";
392  return GlCaps::setError();
393  // TODO: Implement manual fallback (or replace everything with glGenerateMipmap for new versions of GL).
394  }
395  }
396 
397  THEA_GL_OK_OR_RETURN(0)
398 
399  if (THEA_GL_SUPPORTS(ARB_shadow))
400  {
401  if (options->depthReadMode() == Options::DepthReadMode::NORMAL)
402  {
403  glTexParameteri(gl_target, GL_DEPTH_TEXTURE_MODE, GL_INTENSITY);
404  glTexParameteri(gl_target, GL_TEXTURE_COMPARE_MODE_ARB, GL_NONE);
405  }
406  else
407  {
408  glTexParameteri(gl_target, GL_DEPTH_TEXTURE_MODE, GL_INTENSITY);
409  glTexParameteri(gl_target, GL_TEXTURE_COMPARE_MODE_ARB, GL_COMPARE_R_TO_TEXTURE_ARB);
410  glTexParameteri(gl_target, GL_TEXTURE_COMPARE_FUNC_ARB,
411  (options->depthReadMode() == Options::DepthReadMode::LEQUAL) ? GL_LEQUAL : GL_GEQUAL);
412  }
413  }
414  else if (options->depthReadMode() != Options::DepthReadMode::NORMAL)
415  { THEA_ERROR << getName() << ": Comparison-based depth read modes are not supported"; return GlCaps::setError(); }
416 
417  THEA_GL_OK_OR_RETURN(0)
418  return true;
419 }

461 int8
462 GlTexture::updateSubImage(IImage const * image, int64 src_x, int64 src_y, int64 src_z, int64 src_width, int64 src_height,
463  int64 src_depth, int64 dst_x, int64 dst_y, int64 dst_z, int32 face)
464 {
465  if (!image || !image->isValid())
466  { THEA_ERROR << getName() << ": Cannot update texture from null or invalid image"; return GlCaps::setError(); }
467 
468  if (src_x < 0 || src_y < 0 || src_z < 0
469  || src_x + src_width > image->getWidth()
470  || src_y + src_height > image->getHeight()
471  || src_z + src_depth > image->getDepth())
472  { THEA_ERROR << getName() << ": All or part of subimage lies outside source image boundaries"; return GlCaps::setError(); }
473 
474  if (dst_x < 0 || dst_y < 0 || dst_z < 0
475  || dst_x + src_width > width || dst_y + src_height > height || dst_z + src_depth > depth)
476  { THEA_ERROR << getName() << ": All or part of subimage lies outside texture boundaries"; return GlCaps::setError(); }
477 
478  Format const * bytes_format = TextureFormat::fromImageType(IImage::Type(image->getType()));
479 
480  { GlScope scope(GL_TEXTURE_BIT | GL_ENABLE_BIT); // Can we do without ENABLE_BIT? The doc is unclear.
481  GlClientScope client_scope(GL_CLIENT_PIXEL_STORE_BIT);
482 
483  glEnable(gl_target);
484  glBindTexture(gl_target, gl_id);
485  THEA_GL_OK_OR_RETURN(0)
486 
487  if (!GlTextureInternal::setDefaultUnpackingOptions(image->getRowAlignment())) return GlCaps::setError();
488  glPixelStorei(GL_UNPACK_ROW_LENGTH, image->getWidth());
489  glPixelStorei(GL_UNPACK_SKIP_ROWS, src_y);
490  glPixelStorei(GL_UNPACK_SKIP_PIXELS, src_x);
491 
492  switch (gl_target)
493  {
494  case GL_TEXTURE_1D:
495  glTexSubImage1D(gl_target, 0, dst_x, (GLsizei)src_width, bytes_format->openGlBaseFormat(),
496  bytes_format->openGlDataFormat(), image->getData());
497  break;
498 
499  case GL_TEXTURE_2D:
500  case GL_TEXTURE_RECTANGLE_ARB:
501  glTexSubImage2D(gl_target, 0, dst_x, dst_y, (GLsizei)src_width, (GLsizei)src_height, bytes_format->openGlBaseFormat(),
502  bytes_format->openGlDataFormat(), image->getData());
503  break;
504 
505  case GL_TEXTURE_3D:
506 #ifdef GL_VERSION_1_2 // Apple omits the EXT extension after moving 3D textures to core
507  glTexSubImage3D
508 #else
509  glTexSubImage3DEXT
510 #endif
511  (gl_target, 0, dst_x, dst_y, dst_z, (GLsizei)src_width, (GLsizei)src_height, (GLsizei)src_depth,
512  bytes_format->openGlBaseFormat(), bytes_format->openGlDataFormat(), image->getData());
513  break;
514 
515  default: // GL_TEXTURE_CUBE_MAP_ARB
516  glTexSubImage2D(toGlCubeMapFace(face), 0, dst_x, dst_y, (GLsizei)src_width, (GLsizei)src_height,
517  bytes_format->openGlBaseFormat(), bytes_format->openGlDataFormat(), image->getData());
518  }
519  THEA_GL_OK_OR_RETURN(0)
520  }
521 
522  return true;

526 GlTexture::getImage(IImage * image, int32 face) const
