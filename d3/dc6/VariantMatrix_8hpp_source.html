203  // Functions from Serializable
204  void read(BinaryInputStream & input, Codec const & codec = CodecAuto(), bool read_block_header = false)
205  {
206  if (codec == CodecAuto())
207  {
208  auto which = input.readMatrix(read_block_header, dense, sparse);
209  type = (which == 0 ? Type::DENSE : Type::SPARSE);
210  }
211  else
212  {
213  // Decide based on the type of codec supplied
214  if (dynamic_cast< MatrixCodec<Dense> const * >(&codec))
215  {
216  input.readMatrix(read_block_header, dense, codec);
217  type = Type::DENSE;
218  }
219  else if (dynamic_cast< MatrixCodec<Sparse> const * >(&codec))
220  {
221  input.readMatrix(read_block_header, sparse, codec);
222  type = Type::SPARSE;
223  }
224  else
225  throw Error("VariantMatrix: Codec is not valid for either the dense or sparse matrix types");
226  }
227  }
