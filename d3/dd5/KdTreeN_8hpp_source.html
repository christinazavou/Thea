474  template <typename InputIterator>
475  void init(InputIterator begin, InputIterator end, intx max_depth_ = -1, intx max_elems_per_leaf_ = -1,
476  bool save_memory = false, bool deallocate_previous_memory = true)
477  {
478  clear(deallocate_previous_memory);
479 
480  if (deallocate_previous_memory)
481  {
482  for (InputIterator ii = begin; ii != end; ++ii, ++num_elems)
483  if (elementPassesFilters(*ii))
484  elems.push_back(*ii);
485  }
486  else
487  {
488  size_t max_new_elems = (size_t)std::distance(begin, end);
489  bool resized = false;
490  if (max_new_elems > elems.size())
491  {
492  if (filters.empty())
493  elems.resize((size_t)std::ceil(1.2 * max_new_elems)); // add a little more space to avoid future reallocs
494  else
495  elems.clear(); // we don't know how many elements will pass the filter
496 
497  resized = true;
498  }
499 
500  if (filters.empty())
501  {
502  std::copy(begin, end, elems.begin());
503  num_elems = (intx)max_new_elems;
504  }
505  else
506  {
507  if (resized)
508  {
509  for (InputIterator ii = begin; ii != end; ++ii)
510  if (elementPassesFilters(*ii))
511  {
512  elems.push_back(*ii);
513  ++num_elems;
514  }
515  }
516  else
517  {
518  typename ElementArray::iterator ei = elems.begin();
519  for (InputIterator ii = begin; ii != end; ++ii)
520  if (elementPassesFilters(*ii))
521  {
522  *(ei++) = *ii;
523  ++num_elems;
524  }
525  }
526  }
527  }
528 
529  if (num_elems <= 0)
530  return;
531 
532  static intx const DEFAULT_MAX_ELEMS_IN_LEAF = 10;
533  max_elems_per_leaf = max_elems_per_leaf_ < 0 ? DEFAULT_MAX_ELEMS_IN_LEAF : max_elems_per_leaf_;
534 
535  // The fraction of elements held by the larger node at each split is 0.5
536  static double const SPLIT_FRACTION = 0.5;
537  intx est_depth = Math::binaryTreeDepth(num_elems, max_elems_per_leaf, SPLIT_FRACTION);
538  max_depth = max_depth_;
539  if (max_depth < 0)
540  max_depth = est_depth;
541  else if (max_depth < est_depth)
542  est_depth = max_depth;
543 
544  // THEA_CONSOLE << "KdTreeN: max_depth = " << max_depth << ", est_depth = " << est_depth;
545 
546  // Each index is stored at most once at each level
547  size_t BUFFER_SAFETY_MARGIN = 10;
548  size_t index_buffer_capacity = num_elems + BUFFER_SAFETY_MARGIN;
549  if (!save_memory)
550  index_buffer_capacity *= (size_t)(1 + est_depth); // reserve space for all levels at once
551 
552  if (deallocate_previous_memory || index_buffer_capacity > 1.3 * index_pool.getBufferCapacity())
553  {
554  // THEA_CONSOLE << "KdTreeN: Resizing index pool: old buffer capacity = " << index_pool.getBufferCapacity()
555  // << ", new buffer capacity = " << index_buffer_capacity;
556  index_pool.init(index_buffer_capacity);
557  }
558 
559  // Assume a complete, balanced binary tree upto the estimated depth to guess the number of leaf nodes
560  size_t node_buffer_capacity = (size_t)(1 << est_depth) + BUFFER_SAFETY_MARGIN;
561  if (deallocate_previous_memory || node_buffer_capacity > 1.3 * node_pool.getBufferCapacity())
562  {
563  // THEA_CONSOLE << "KdTreeN: Resizing node pool: old buffer capacity = " << node_pool.getBufferCapacity()
564  // << ", new buffer capacity = " << node_buffer_capacity;
565  node_pool.init(node_buffer_capacity);
566  }
567 
568  // Create the root node
569  root = node_pool.alloc(1);
570  root->init(0);
571  num_nodes = 1;
572 
573  // THEA_CONSOLE << "Allocated root " << root << " from mempool " << &node_pool;
574 
575  root->num_elems = num_elems;
576  root->elems = index_pool.alloc(root->num_elems);
577 
578  AxisAlignedBoxT elem_bounds;
579  for (size_t i = 0; i < (size_t)num_elems; ++i)
580  {
581  root->elems[i] = i;
582 
583  BoundedTraitsT::getBounds(elems[i], elem_bounds);
584  root->bounds.merge(elem_bounds);
585  }
586 
587  // Expand the bounding box slightly to handle numerical error
588  root->bounds.scaleCentered(BOUNDS_EXPANSION_FACTOR);
589 
590  if (save_memory)
591  {
592  // Estimate the maximum number of indices that will need to be held in the scratch pool at any time during depth-first
593  // traversal with earliest-possible deallocation. This is
594  //
595  // #elements * (sum of series 1 + 1 + SPLIT_FRACTION + SPLIT_FRACTION^2 + ... + SPLIT_FRACTION^(est_depth - 1))
596  // <= #elements * (1 + 1 / (1 - SPLIT_FRACTION))
597  //
598  size_t est_max_path_indices = (size_t)(num_elems * (1 + 1 / (1 - SPLIT_FRACTION)));
599  // THEA_CONSOLE << "KdTreeN: Estimated maximum number of indices on a single path = " << est_max_path_indices;
600 
601  // Create a temporary pool for scratch storage
602  IndexPool tmp_index_pool;
603  tmp_index_pool.init(est_max_path_indices + BUFFER_SAFETY_MARGIN);
604 
605  createTree(root, true, &tmp_index_pool, &index_pool);
606  }
607  else
608  createTree(root, false, &index_pool, nullptr);
609 
610  invalidateBounds();
611  }
