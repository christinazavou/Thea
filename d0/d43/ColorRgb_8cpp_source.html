193 ColorRgb::fromHSV(Vector3 const & hsv)
194 {
195  debugAssertM((hsv[0] <= 1.0f && hsv[0] >= 0.0f)
196  && (hsv[1] <= 1.0f && hsv[1] >= 0.0f)
197  && (hsv[2] <= 1.0f && hsv[2] >= 0.0f), "ColorRgb: H, S, V must be in [0, 1]");
198 
199  int const i = std::min(5, (int)std::floor(6.0 * hsv[0]));
200  Real const f = 6.0f * hsv[0] - i;
201  Real const m = hsv[2] * (1.0f - (hsv[1]));
202  Real const n = hsv[2] * (1.0f - (hsv[1] * f));
203  Real const k = hsv[2] * (1.0f - (hsv[1] * (1 - f)));
204 
205  switch (i)
206  {
207  case 0: return ColorRgb(hsv[2], k, m);
208  case 1: return ColorRgb(n, hsv[2], m);
209  case 2: return ColorRgb(m, hsv[2], k);
210  case 3: return ColorRgb(m, n, hsv[2]);
211  case 4: return ColorRgb(k, m, hsv[2]);
212  case 5: return ColorRgb(hsv[2], m, n);
213 
214  default: debugAssertM(false, "ColorRgb: Fell through switch when attempting conversion from HSV");
215  }
216 
217  return ColorRgb::black();
218 }

220 Vector3
221 ColorRgb::toHSV() const
222 {
223  debugAssertM((c[0] <= 1.0f && c[0] >= 0.0f)
224  && (c[1] <= 1.0f && c[1] >= 0.0f)
225  && (c[2] <= 1.0f && c[2] >= 0.0f), "ColorRgb: R, G, B must be in [0, 1]");
226 
227  Vector3 hsv = Vector3::Zero();
228  hsv[2] = std::max(std::max(c[0], c[1]), c[2]);
229 
230  if (Math::fuzzyEq(hsv[2], (Real)0))
231  return hsv;
232 
233  Real const x = std::min(std::min(c[0], c[1]), c[2]);
234  hsv[1] = (hsv[2] - x) / hsv[2];
235 
236  if (Math::fuzzyEq(hsv[1], (Real)0))
237  return hsv;
238 
239  Vector3 rgbN;
240  rgbN[0] = (hsv[2] - c[0]) / (hsv[2] - x);
241  rgbN[1] = (hsv[2] - c[1]) / (hsv[2] - x);
242  rgbN[2] = (hsv[2] - c[2]) / (hsv[2] - x);
243 
244  if (c[0] == hsv[2]) // Note: from the max we know that it exactly equals one of the three
245  {
246  hsv[0] = (c[1] == x) ? 5.0f + rgbN[2] : 1.0f - rgbN[1];
247  }
248  else if (c[1] == hsv[2])
249  {
250  hsv[0] = (c[2] == x) ? 1.0f + rgbN[0] : 3.0f - rgbN[2];
251  }
252  else
253  {
254  hsv[0] = (c[0] == x) ? 3.0f + rgbN[1] : 5.0f - rgbN[0];
255  }
256 
257  hsv[0] /= 6.0f;
258 
259  return hsv;
260 }
