29 int64
30 ArpackEigenSolver::solve(IMatrix<float64> const * m, int8 compute_eigenvectors, int64 num_requested_eigenpairs,
31  IOptions const * options)
32 {
33  if (!m || !Math::isSquare(*m))
34  {
35  THEA_ERROR << getName() << ": Operator matrix must be square and non-null";
36  return -1;
37  }
38 
39  eigenvalues [0].clear(); eigenvalues [1].clear();
40  eigenvectors[0].clear(); eigenvectors[1].clear();
41 
42  ndims = m->rows();
43  if (ndims <= 0)
44  {
45  THEA_WARNING << getName() << ": Operator matrix has zero dimensions";
46  return 0;
47  }
48 
49  if (num_requested_eigenpairs < 0)
50  num_requested_eigenpairs = ndims;
51 
52  char const * DEFAULT_WHICH = "LM";
53  int32 const DEFAULT_NCV = std::min(2 * num_requested_eigenpairs + 1, ndims - 1);
54  float64 const DEFAULT_TOL = std::max(1e-7, 5e-9 * ndims);
55  int32 const DEFAULT_MAXIT = 100 * num_requested_eigenpairs;
56  int32 const DEFAULT_SIGMA = 0;
57 
58  std::string which_s = options ? options->getString("which", DEFAULT_WHICH) : DEFAULT_WHICH;
59  char which[3];
60  std::strncpy(which, which_s.c_str(), 2); // make a copy since ARPACK++ doesn't const-protect this string
61  which[2] = 0;
62 
63  int32 ncv = options ? (int32)options->getInteger("ncv", DEFAULT_NCV ) : DEFAULT_NCV;
64  float64 tol = options ? options->getFloat ("tol", DEFAULT_TOL ) : DEFAULT_TOL;
65  int32 maxit = options ? (int32)options->getInteger("maxit", DEFAULT_MAXIT) : DEFAULT_MAXIT;
66 
67  float64 sigma = 0;
68  int8 shift_invert = options && options->hasOption("sigma");
69  if (shift_invert)
70  sigma = options->getFloat("sigma", DEFAULT_SIGMA);
71 
72  THEA_DEBUG << getName() << ": nev = " << num_requested_eigenpairs << ", which = " << which << ", ncv = " << ncv
73  << ", tol = " << tol << ", maxit = " << maxit;
74 
75  if (m->asSparse() && m->asSparse()->asCompressed())
76  return solveSparse(*m->asSparse()->asCompressed(), num_requested_eigenpairs, shift_invert, sigma, which, ncv, tol, maxit);
77  else if (m->asAddressable() && m->asAddressable()->asDense())
78  return solveDense(*m->asAddressable()->asDense(), num_requested_eigenpairs, shift_invert, sigma, which, ncv, tol, maxit);
79  else
80  {
81  THEA_ERROR << getName() << ": Unsupported matrix format";
82  return -1;
83  }
84 }

87 ArpackEigenSolver::getEigenvalue(int64 i, float64 * re, float64 * im) const
88 {
89  if (i < 0 || i >= (int64)eigenvalues[0].size())

97  THEA_ERROR << getName() << ": Index of eigenvalue out of range";
98  return false;
99  }
100 
101  if (re) *re = eigenvalues[0][(size_t)i];
