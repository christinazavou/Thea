291 DisplayMesh::Face
292 DisplayMesh::addFace(int num_vertices, intx const * face_vertex_indices_, intx source_face_index)
293 {
294  if (num_vertices < 3)
295  {
296  THEA_DEBUG << getName() << ": Skipping face -- too few vertices (" << num_vertices << ')';
297  return Face();
298  }
299 
300  if (num_vertices == 3)
301  return Face(this, 3, true,
302  addTriangle(face_vertex_indices_[0], face_vertex_indices_[1], face_vertex_indices_[2], source_face_index), 1);
303 
304  if (num_vertices == 4)
305  return Face(this, 4, false,
306  addQuad(face_vertex_indices_[0], face_vertex_indices_[1], face_vertex_indices_[2], face_vertex_indices_[3],
307  source_face_index),
308  1);
309 
310  Polygon3 poly;
311  for (int i = 0; i < num_vertices; ++i)
312  {
313  intx vi = face_vertex_indices_[i];
314  debugAssertM(vi >= 0 && vi < (intx)vertices.size(), getName() + format(": Vertex index %ld out of bounds", vi));
315 
316  poly.addVertex(vertices[vi], vi);
317  }
318 
319  intx num_tris = poly.triangulate(triangulated_indices);
320  if (num_tris <= 0)
321  return Face();
322 
323  // debugAssertM(num_tris == 3 * (num_vertices - 2),
324  // getName() + format(": Triangulation of polygonal face yielded %l triangles, whereas %l were expected",
325  // num_tris, num_vertices - 2));
326 
327  alwaysAssertM((source_face_index >= 0 && 3 * tri_source_face_indices.size() == tris.size())
328  || (source_face_index < 0 && tri_source_face_indices.empty()),
329  getNameStr() + ": Mesh must have all or no triangle face source indices");
330 
331  intx starting_index = numTriangles();
332  size_t num_tri_verts = (size_t)(3 * num_tris);
333  for (size_t i = 0; i < num_tri_verts; i += 3)
334  {
335  tris.push_back((uint32)triangulated_indices[i ]);
336  tris.push_back((uint32)triangulated_indices[i + 1]);
337  tris.push_back((uint32)triangulated_indices[i + 2]);
338  if (source_face_index >= 0) tri_source_face_indices.push_back(source_face_index);
339  }
340 
341  invalidateGpuBuffers();
342 
343  return Face(this, num_vertices, true, starting_index, num_tris);
344 }

405 void
406 DisplayMesh::computeAveragedVertexNormals()
407 {
408  bool topo_change = (normals.size() != vertices.size());
409 
410  normals.resize(vertices.size());
411  for (size_t i = 0; i < normals.size(); ++i)
412  normals[i] = Vector3::Zero();
413 
414  // TODO: weight normals by face area?
415  Vector3 n;
416  uint32 i0, i1, i2, i3;
417  for (size_t i = 0; i < tris.size(); i += 3)
418  {
419  i0 = tris[i], i1 = tris[i + 1], i2 = tris[i + 2];
420  Vector3 const & v0 = vertices[i0];
421  Vector3 const & v1 = vertices[i1];
422  Vector3 const & v2 = vertices[i2];
423 
424  n = (v2 - v1).cross(v0 - v1).normalized();
425  normals[i0] += n;
426  normals[i1] += n;
427  normals[i2] += n;
428  }
429 
430  for (size_t i = 0; i < quads.size(); i += 4)
431  {
432  i0 = quads[i], i1 = quads[i + 1], i2 = quads[i + 2], i3 = quads[i + 3];
433  Vector3 const & v0 = vertices[i0];
434  Vector3 const & v1 = vertices[i1];
435  Vector3 const & v2 = vertices[i2];
436 
437  n = (v2 - v1).cross(v0 - v1).normalized();
438  normals[i0] += n;
439  normals[i1] += n;
440  normals[i2] += n;
441  normals[i3] += n;
442  }
443 
444  for (size_t i = 0; i < normals.size(); ++i)
445  normals[i] = normals[i].normalized();
446 
447  invalidateGpuBuffers(topo_change ? BufferId::ALL : BufferId::VERTEX_NORMAL);
448 }

459 void
460 DisplayMesh::updateEdges()
461 {
462  edges.clear();
463 
464  if (wireframe_enabled)
465  {
466  typedef std::pair<uint32, uint32> Edge;
467  typedef UnorderedSet<Edge> EdgeSet;
468 
469  EdgeSet added_edges;
470  Edge edge;
471 
472  for (size_t i = 0; i < tris.size(); i += 3)
473  for (size_t j = 0; j < 3; ++j)
474  {
475  uint32 ei0 = tris[i + j];
476  uint32 ei1 = tris[j == 2 ? i : (i + j + 1)];
477 
478  // Order so lower index is first, since we're considering edges to be undirected
479  if (ei0 < ei1)
480  edge = Edge(ei0, ei1);
481  else
482  edge = Edge(ei1, ei0);
483 
484  EdgeSet::iterator existing = added_edges.find(edge);
485  if (existing == added_edges.end())
486  {
487  edges.push_back(ei0);
488  edges.push_back(ei1);
489  added_edges.insert(edge);
490  }
491  }
492 
493  for (size_t i = 0; i < quads.size(); i += 4)
494  for (size_t j = 0; j < 4; ++j)
495  {
496  uint32 ei0 = quads[i + j];
497  uint32 ei1 = quads[j == 3 ? i : (i + j + 1)];
498 
499  // Order so lower index is first, since we're considering edges to be undirected
500  if (ei0 < ei1)
501  edge = Edge(ei0, ei1);
502  else
503  edge = Edge(ei1, ei0);
504 
505  EdgeSet::iterator existing = added_edges.find(edge);
506  if (existing == added_edges.end())
507  {
508  edges.push_back(ei0);
509  edges.push_back(ei1);
510  added_edges.insert(edge);
511  }
512  }
513  }
514 
515  THEA_DEBUG << getName() << ": Mesh has " << edges.size() / 2 << " edges";
516 }

536  new_vertices.push_back(vertices[i2]);
537 
538  if (!normals.empty())
539  {
540  new_normals.push_back(normals[i0]);
541  new_normals.push_back(normals[i1]);

557  }

560  tris[i + 1] = new_vindex + 1;
561  tris[i + 2] = new_vindex + 2;
562  }
563 
564  for (size_t i = 0; i < quads.size(); i += 4)

579  {

584  }
585

587  {
588  new_colors.push_back(colors[i0]);

659  intx num_bytes = vertex_bytes + normal_bytes + color_bytes + texcoord_bytes + PADDING;
660 #else
661  intx tri_bytes = !tris.empty() ? 4 * (intx)tris.size() + PADDING : 0; // uint32
662  intx quad_bytes = !quads.empty() ? 4 * (intx)quads.size() + PADDING : 0; // uint32
663  intx edge_bytes = !edges.empty() ? 4 * (intx)edges.size() + PADDING : 0; // uint32
664 
665  intx num_bytes = vertex_bytes + normal_bytes + color_bytes + texcoord_bytes + tri_bytes + quad_bytes + edge_bytes + PADDING;
666 #endif
667 
668  if (buf_pool)
669  {
670  if (buf_pool->getCapacity() <= num_bytes || buf_pool->getCapacity() > (intx)(1.5 * num_bytes))
671  {
672  render_system.destroyBufferPool(buf_pool);
673 
674  std::string pool_name = getNameStr() + " buffer pool";
675  buf_pool = render_system.createBufferPool(pool_name.c_str(), num_bytes, IBufferPool::Usage::WRITE_OCCASIONALLY, true);
676  if (!buf_pool) return false;
677  }
678  // Else no need to reset buf_pool, we've done it above
679  }
680  else
681  {
682  std::string pool_name = getNameStr() + " buffer pool";
683  buf_pool = render_system.createBufferPool(pool_name.c_str(), num_bytes, IBufferPool::Usage::WRITE_OCCASIONALLY, true);
684  if (!buf_pool) return false;
685  }
686 
687  if (!vertices.empty() && !(vertices_buf = buf_pool->createBuffer(vertex_bytes)) ) return false;
688  if (hasNormals() && !(normals_buf = buf_pool->createBuffer(normal_bytes)) ) return false;
689  if (hasColors() && !(colors_buf = buf_pool->createBuffer(color_bytes)) ) return false;
690  if (hasTexCoords() && !(texcoords_buf = buf_pool->createBuffer(texcoord_bytes))) return false;
691 
692 #ifndef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
693  if (!tris.empty() && !(tris_buf = buf_pool->createBuffer(tri_bytes))) return false;
694  if (!quads.empty() && !(quads_buf = buf_pool->createBuffer(quad_bytes))) return false;
695  if (!edges.empty() && !(edges_buf = buf_pool->createBuffer(edge_bytes))) return false;
696 
697  if (!tris.empty() && !tris_buf->updateIndices (0, (int64)tris.size(), NumericType::UINT32, &tris[0]) ) return false;
698  if (!quads.empty() && !quads_buf->updateIndices(0, (int64)quads.size(), NumericType::UINT32, &quads[0])) return false;
699  if (!edges.empty() && !edges_buf->updateIndices(0, (int64)edges.size(), NumericType::UINT32, &edges[0])) return false;
700 #endif
701 
702  if (!vertices.empty() && !vertices_buf->updateAttributes(0, (int64)vertices.size(), 3, NumericType::REAL, &vertices[0]))
703  return false;
704 
705  if (hasNormals() && !normals_buf->updateAttributes(0, (int64)normals.size(), 3, NumericType::REAL, &normals[0]))
706  return false;
707 
708  if (hasColors() && !colors_buf->updateAttributes(0, (int64)colors.size(), 4, NumericType::REAL, &colors[0]))
709  return false;
710 
711  if (hasTexCoords() && !texcoords_buf->updateAttributes(0, (int64)texcoords.size(), 2, NumericType::REAL, &texcoords[0]))
712  return false;
713  }
714  else
715  {
716  if (!gpuBufferIsValid(BufferId::VERTEX_POSITION) && !vertices.empty()
717  && !vertices_buf->updateAttributes(0, (int64)vertices.size(), 3, NumericType::REAL, &vertices[0])) return false;
718 
719  if (!gpuBufferIsValid(BufferId::VERTEX_NORMAL) && hasNormals()
720  && !normals_buf->updateAttributes(0, (int64)normals.size(), 3, NumericType::REAL, &normals[0])) return false;
721 
722  if (!gpuBufferIsValid(BufferId::VERTEX_COLOR) && hasColors()
723  && !colors_buf->updateAttributes(0, (int64)colors.size(), 4, NumericType::REAL, &colors[0])) return false;
724 
725  if (!gpuBufferIsValid(BufferId::VERTEX_TEXCOORD) && hasTexCoords()
726  && !texcoords_buf->updateAttributes(0, (int64)texcoords.size(), 2, NumericType::REAL, &texcoords[0])) return false;
727  }
728 
729  allGpuBuffersAreValid();
730 
731  return true;

735 DisplayMesh::draw(IRenderSystem * render_system, IRenderOptions const * options) const
736 {
737  if (!render_system) { THEA_ERROR << getName() << ": Can't display mesh on a null rendersystem"; return false; }
738  if (!options) options = RenderOptions::defaults();
739 
740  if (options->drawEdges() && !wireframe_enabled)
741  { THEA_ERROR << getName() << ": Can't draw mesh edges when wireframe mode is disabled"; return false; }
742 
743  if (!const_cast<DisplayMesh *>(this)->uploadToGraphicsSystem(*render_system)) return false;
744 
745  if (!vertices_buf) return true;
746  if (!options->drawFaces() && !options->drawEdges()) return true;
747  if (!options->drawFaces() && !edges_buf) return true;
748  if (!options->drawEdges() && !tris_buf && !quads_buf) return true;
749 
750  render_system->beginIndexedPrimitives();
751 
752  render_system->setVertexBuffer(vertices_buf);
753  if (options->sendNormals() && normals_buf) render_system->setNormalBuffer(normals_buf);
754  if (options->sendColors() && colors_buf) render_system->setColorBuffer(colors_buf);
755  if (options->sendTexCoords() && texcoords_buf) render_system->setTexCoordBuffer(0, texcoords_buf);
756 
757  if (options->drawFaces())
758  {
759  if (options->drawEdges())
760  {
761  render_system->pushShapeFlags();
762  render_system->setPolygonOffset(true, 2);
763  }
764 
765 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
766  if (!tris.empty()) render_system->sendIndices(IRenderSystem::Primitive::TRIANGLES, (int64)tris.size(), &tris[0]);
767  if (!quads.empty()) render_system->sendIndices(IRenderSystem::Primitive::QUADS, (int64)quads.size(), &quads[0]);
768 #else
769  if (!tris.empty())
770  {
771  render_system->setIndexBuffer(tris_buf);
772  render_system->sendIndicesFromBuffer(IRenderSystem::Primitive::TRIANGLES, 0, (int64)tris.size());
773  }
774 
775  if (!quads.empty())
776  {
777  render_system->setIndexBuffer(quads_buf);
778  render_system->sendIndicesFromBuffer(IRenderSystem::Primitive::QUADS, 0, (int64)quads.size());
779  }
780 #endif
781 
782  if (options->drawEdges())
783  render_system->popShapeFlags();
784  }
785 
786  if (options->drawEdges())
787  {
788  render_system->pushShader();
789  render_system->pushColorFlags();
790  render_system->pushTextures();
791 
792  render_system->setShader(nullptr);
793  render_system->setColorBuffer(nullptr);
794  render_system->setTexCoordBuffer(0, nullptr);
795  render_system->setNormalBuffer(nullptr);
796  render_system->setColor(options->edgeColor()); // set default edge color (TODO: handle per-edge colors)
797  render_system->setTexture(0, nullptr);
798 
799 #ifdef THEA_DISPLAY_MESH_NO_INDEX_ARRAY
800  if (!edges.empty()) render_system->sendIndices(IRenderSystem::Primitive::LINES, (int64)edges.size(), &edges[0]);
801 #else
802  if (!edges.empty())
803  {
804  render_system->setIndexBuffer(edges_buf);
805  render_system->sendIndicesFromBuffer(IRenderSystem::Primitive::LINES, 0, (int64)edges.size());
806  }
807 #endif
808 
809  render_system->popTextures();
810  render_system->popColorFlags();
811  render_system->popShader();
812  }
813 
814  render_system->endIndexedPrimitives();
815 
816  if (char const * err = render_system->getLastError())
817  { THEA_ERROR << getName() << ": Rendering error (" << err << ')'; return false; }
818 
819  return true;
820 }
